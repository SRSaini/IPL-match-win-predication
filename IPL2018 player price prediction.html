#Fitting PCA with training data
pca = PCA()
X_reduced_train = pca.fit_transform(scale(X))
n = len(X_reduced_train)
# 11-fold CV, with shuffle
kf_11 = KFold(n_splits=11, shuffle=False, random_state=1)
regr = LinearRegression()
mse = []
# Calculate MSE with only the intercept (no principal components in regression)
score = -1*cross_val_score(regr, np.ones((n,1)), scale(y), cv=kf_11, scoring='neg_mean_squared_error').mean()    
mse.append(score)
# Calculate MSE using CV for the 19 principle components, adding one component at the time.
for i in np.arange(1, 30):
    score = -1*cross_val_score(regr, X_reduced_train[:,:i], scale(y), cv=kf_11, scoring='neg_mean_squared_error').mean()
    mse.append(score)
plt.plot(np.array(mse), '-v')
plt.xlabel('Number of principal components in regression')
plt.ylabel('MSE for PCA')
plt.title('price')
plt.xlim(xmin=-1)
plt.show()
